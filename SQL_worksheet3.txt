1. create table customers (customernumber int NOT NULL,customername varchar(30),contactlastname varchar(20),contactfirstname varchar(20),phone int,addressline1 varchar(30),addressline2 varchar(30),city varchar(20),state varchar(20),postalcode int,country varchar(20),salesrepnum int,creditlimit int,PRIMARY KEY(customernumber),FOREIGN KEY(salesrepnum) REFERENCES employees(employeenumber));

2. create table orders (ordernumber int NOT NULL,ordr_date date,required_date date,shipped_date date,
status varchar(10),comments varchar(30),customernumber int, PRIMARY KEY(ordernumber),
FOREIGN KEY(customernumber) REFERENCES customers(customernumber));
